<HowToGuide
  problem="You need a button with custom styling that matches your design requirements, but the default variants don't fit your needs."
  solution="Use existing variants, extend with Tailwind classes, or create a new variant using class-variance-authority."
  steps={[
    {
      title: "Use existing variants",
      description: "The Button component supports multiple built-in variants. Try these first before creating custom styles.",
      code: `<Button variant="primary">Primary</Button>
<Button variant="secondary">Secondary</Button>
<Button variant="outline">Outline</Button>`
    },
    {
      title: "Add custom Tailwind classes",
      description: "You can extend button styles by passing additional Tailwind classes through the className prop.",
      code: `<Button className="bg-gold text-void rounded-lg">
  Custom Button
</Button>`,
      warning: "Avoid overriding core design tokens. Use semantic colors when possible."
    },
    {
      title: "Create a new variant",
      description: "For reusable custom variants, extend the Button component using class-variance-authority (CVA).",
      code: `// components/CustomButton.tsx
import { Button } from '@mythic/design-system/components'
import { cva } from 'class-variance-authority'

const customButtonVariants = cva(
  'base-classes',
  {
    variants: {
      variant: {
        custom: 'bg-gold text-void hover:bg-ember',
      },
    },
  }
)

export function CustomButton({ variant, ...props }) {
  return <Button className={customButtonVariants({ variant })} {...props} />
}`
    }
  ]}
/>
