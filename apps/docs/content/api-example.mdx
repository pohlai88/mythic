# API Reference Example

This page demonstrates the API Reference component for documenting REST APIs.

<APIReference
  endpoints={[
    {
      method: 'GET',
      path: '/api/users',
      description: 'Retrieve a list of all users',
      parameters: [
        {
          name: 'page',
          type: 'number',
          required: false,
          description: 'Page number for pagination',
        },
        {
          name: 'limit',
          type: 'number',
          required: false,
          description: 'Number of items per page',
        },
      ],
      response: {
        status: 200,
        example: `{
  "users": [
    {
      "id": 1,
      "name": "John Doe",
      "email": "john@example.com"
    }
  ],
  "total": 100,
  "page": 1
}`,
      },
    },
    {
      method: 'POST',
      path: '/api/users',
      description: 'Create a new user',
      parameters: [
        {
          name: 'name',
          type: 'string',
          required: true,
          description: "User's full name",
        },
        {
          name: 'email',
          type: 'string',
          required: true,
          description: "User's email address",
        },
      ],
      response: {
        status: 201,
        example: `{
  "id": 123,
  "name": "Jane Doe",
  "email": "jane@example.com",
  "createdAt": "2024-12-19T10:00:00Z"
}`,
      },
    },
  ]}
/>

---

## Usage

Import the component in your MDX file:

```mdx
import { APIReference } from '@/components'

<APIReference endpoints={[...]} />
```
